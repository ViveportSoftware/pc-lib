    Example

    // The VIVERSE VR system might not be ready when the script is loaded.
    button.active = false;
    button.on('click', () => {
       xrService.start(XrTypes.SessionTypes.Vr);
    });
    if(xrService.isVrReady){
      button.active = true;
    } else {
      xrService.on('vr:ready', () => {
        button.active = true;
        xrService.off('vr:ready');
      });
